#    pip uninstall torch torchvision torchaudio -y
#    pip install torch==2.5.1 torchvision==0.20.1 --index-url https://download.pytorch.org/whl/cpu
#
# 2) PyG binary deps matching that torch (pick the URL that matches your installed torch):
#    pip install torch-scatter torch-sparse -f https://data.pyg.org/whl/torch-2.5.1+cpu.html
#
# 3) Remaining packages:
#    pip install -r requirements-pka.txt
